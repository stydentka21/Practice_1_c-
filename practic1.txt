#include<iostream>
#include<cmath>
using namespace std;
double calculator(double a, double b, char operation)
{
	switch (operation)
	{
	case('+'):
		return a + b;
	case('-'):
		return a - b;
	case('*'):
		return a * b;
	case('/'):
		return a / b;
	default:
		cout << "Unknown operation";
		return -1;
	}
	return 0;
}
int main()
{
	double a, b;
	char operation;
	cin >> a >> operation>> b;
	cout << calculator(a, b, operation);
	return 0;
}
